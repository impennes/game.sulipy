Indul a Turtle!
Változó értékének megadása 

  import time
  import turtle
  import random

  colors = ['orange', 'red', 'blue', 'yellow', 'green']
  turtle.pen(fillcolor='orange', pencolor='red', pensize=7)
  turtle.pen(fillcolor=random.choice(colors), pencolor=random.choice(colors), pensize=7)
  
  turtle.begin_fill()
  for _ in range(4):
      turtle.forward(150)
      turtle.left(90)
  turtle.end_fill()

  time.sleep(5)
