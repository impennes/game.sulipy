A videóban látható kód 

  import turtle

  turtle.shape("turtle")
  turtle.pen(fillcolor="red", pencolor="yellow", pensize=10)
  turtle.speed(1)
  turtle.begin_fill()
  turtle.forward(150)
  turtle.left(90)
  turtle.forward(150)
  turtle.goto(0,0)
  turtle.left(90)
  turtle.forward(150)
  turtle.left(90)
  turtle.forward(150)
  turtle.home()
  turtle.end_fill()
